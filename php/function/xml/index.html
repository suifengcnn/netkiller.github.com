3.15.1.3. saveHTML

下面的例子是从某个网站扣取一个HTML块的例子

		

$url = "http://netkiller.github.io/journal/index.html";
$html =  curl($url);
libxml_use_internal_errors(true);

$dom = new DOMDocument();
$dom->loadHTML($html);
$xpath = new DOMXPath($dom);

$xml = $xpath->query('//div[@class="section"]');

$xhtml = $dom->saveHTML($xml->item(0));

print_r($xhtml);
